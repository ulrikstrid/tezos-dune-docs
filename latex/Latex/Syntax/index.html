Module Latex.Syntax

type t = {
title : string;
sections : section list;
}
and section =
| Section of string * section_content list
and section_content =
| Text of text
| Figure of caption * image
| Table of table
and text = blob list
and blob =
| Text_blob of style * string
| Inline_math_blob of string
and style =
| Normal
| Emph
| Bold
and image = {
filename : string;
size : image_size option;
}
and image_size =
| Width_cm of int
and caption = text
and table = table_spec * row list
and row =
| Row of text list
| Hline
and table_spec = spec list
and spec =
| L
| C
| R
| Vbar
val spec_width : table_spec -> int
val row_width : row -> int option
val map_string : (string -> string) -> t -> t
val map_string_section : (string -> string) -> section -> section
val map_string_section_content : (string -> string) -> section_content -> section_content
val map_text : (string -> string) -> text -> text
val map_string_blob : (string -> string) -> blob -> blob
val map_string_row : (string -> string) -> row -> row
