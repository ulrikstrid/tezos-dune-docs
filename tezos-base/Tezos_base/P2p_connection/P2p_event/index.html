type t =
| Too_few_connections
| Too_many_connections
| New_point of P2p_point.Id.t
| New_peer of P2p_peer_id.t

Garbage collection of known point table has been triggered.

*)

*)

*)

*)

*)

source : P2p_peer_id.t;
}		(*

*)

source : P2p_peer_id.t;

| Swap_failure of {
